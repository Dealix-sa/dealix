#### Step 2: Create API Endpoint for Approvals

**File:** Create `company/api/approvals_api.py` (NEW)

```python
#!/usr/bin/env python3
"""Simple API endpoint for processing approval queue actions."""

import json
import os
from pathlib import Path
from datetime import datetime
from company.automation import WhatsAppAutomation, MoyasarPayments

RUNTIME_DIR = Path(__file__).resolve().parents[2] / "company" / "runtime"

def handle_approve(item_id: str) -> dict:
    """Approve and send an item from approval queue."""
    
    # Load approval queue
    queue_file = RUNTIME_DIR / "approval_queue.json"
    with open(queue_file, "r") as f:
        queue = json.load(f)
    
    # Find item
    item = None
    for i in queue.get("items", []):
        if i.get("id") == item_id:
            item = i
            break
    
    if not item:
        return {"status": "error", "error": "Item not found"}
    
    # Send based on type
    wa = WhatsAppAutomation()
    
    if item.get("type") == "objection_response":
        result = wa.send_message(
            recipient_phone=item.get("phone"),
            message_ar=item.get("draft_ar"),
            message_en=item.get("draft_en")
        )
        
        if result.get("status") == "sent":
            return {
                "status": "approved",
                "phone": item.get("phone"),
                "message_id": result.get("message_id")
            }
        else:
            return {"status": "error", "error": result.get("error")}
    
    elif item.get("type") == "followup_reminder":
        # If scheduled_send is in future, queue for later
        # Otherwise send now
        result = wa.send_message(
            recipient_phone=item.get("phone"),
            message_ar=item.get("draft_ar")
        )
        
        if result.get("status") == "sent":
            return {
                "status": "approved",
                "phone": item.get("phone"),
                "message_id": result.get("message_id")
            }
        else:
            return {"status": "error", "error": result.get("error")}
    
    elif item.get("type") == "payment_link":
        # Generate Moyasar payment link
        moyasar = MoyasarPayments()
        
        result = moyasar.create_payment_link(
            customer_email=item.get("customer_email"),
            customer_phone=item.get("customer_phone"),
            customer_name=item.get("customer_name"),
            description=item.get("description"),
            amount_sar=item.get("amount_sar"),
            pilot_id=item.get("pilot_id")
        )
        
        if result.get("status") == "created":
            # Send payment link via WhatsApp
            message = f"رابط الدفع: {result.get('payment_url')}"
            wa_result = wa.send_message(
                recipient_phone=item.get("customer_phone"),
                message_ar=message
            )
            
            return {
                "status": "approved",
                "payment_url": result.get("payment_url"),
                "invoice_id": result.get("invoice_id"),
                "message_sent": wa_result.get("status") == "sent"
            }
        else:
            return {"status": "error", "error": result.get("error")}
    
    return {"status": "error", "error": "Unknown item type"}

def handle_reject(item_id: str) -> dict:
    """Reject and archive an item."""
    
    queue_file = RUNTIME_DIR / "approval_queue.json"
    archive_file = RUNTIME_DIR / f"approval_queue_archived_{datetime.now().isoformat()}.json"
    
    with open(queue_file, "r") as f:
        queue = json.load(f)
    
    # Find and remove item
    item = None
    remaining = []
    for i in queue.get("items", []):
        if i.get("id") == item_id:
            item = i
        else:
            remaining.append(i)
    
    if not item:
        return {"status": "error", "error": "Item not found"}
    
    # Update queue
    queue["items"] = remaining
    with open(queue_file, "w") as f:
        json.dump(queue, f)
    
    # Archive rejected item
    with open(archive_file, "a") as f:
        f.write(json.dumps({
            "timestamp": datetime.now().isoformat(),
            "action": "rejected",
            "item": item
        }) + "\n")
    
    return {"status": "rejected", "item_id": item_id}
```

#### Step 2: Add Payment Status Monitoring

**Create:** `scripts/check_payment_status.py` (NEW)

```python
#!/usr/bin/env python3
"""Check payment status for all active pilots."""

import json
from pathlib import Path
from company.automation import MoyasarPayments

RUNTIME_DIR = Path(__file__).resolve().parents[2] / "company" / "runtime"
PILOTS_DIR = RUNTIME_DIR / "pilots"

def check_all_payments():
    """Check payment status for all pilots."""
    
    moyasar = MoyasarPayments()
    results = []
    
    for pilot_dir in PILOTS_DIR.glob("pilot_*"):
        payment_file = pilot_dir / "payment_link.json"
        
        if not payment_file.exists():
            continue
        
        with open(payment_file) as f:
            payment_info = json.load(f)
        
        invoice_id = payment_info.get("invoice_id")
        status_result = moyasar.check_payment_status(invoice_id)
        
        # Update status
        payment_info['current_status'] = status_result.get('status')
        payment_info['paid_amount_sar'] = status_result.get('paid_amount_sar')
        payment_info['last_checked'] = datetime.now().isoformat()
        
        with open(payment_file, "w") as f:
            json.dump(payment_info, f)
        
        results.append({
            "pilot_id": pilot_dir.name,
            "status": status_result.get('status'),
            "paid_amount": status_result.get('paid_amount_sar'),
            "amount_due": status_result.get('amount_sar')
        })
    
    # Log results
    log_file = RUNTIME_DIR / f"payment_check_{datetime.now().strftime('%Y-%m-%d')}.json"
    with open(log_file, "w") as f:
        json.dump(results, f)
    
    return results

if __name__ == "__main__":
    results = check_all_payments()
    for r in results:
        status = "✅" if r['status'] == 'paid' else "⏳"
        print(f"{status} {r['pilot_id']}: {r['status']} ({r['paid_amount']}/{r['amount_due']} SAR)")
```
